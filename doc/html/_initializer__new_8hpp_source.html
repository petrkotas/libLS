324 template <typename DataProvider, int dim> int Initializer<DataProvider, dim>::selectBoundaries(Grid<double, dim>* gr) {
325  // chooses which boundaries of the domain have to be initialized
326  // ordering : [left, right, bottom, top, front, back]
327 
328  int selectedBoundaries_int = 0;
329 
330  for (int i = 0; i < dim; ++i) {
331  if ( gr->getLocalGhostedMax(i) <= dataProvider->getMaxX(i) ) {
332  selectedBoundaries_int |= ( 1 << (i*2 + 1) );
333  }
334  if ( gr->getLocalGhostedMin(i) >= dataProvider->getMinX(i) ) {
335  selectedBoundaries_int |= ( 1 << (i*2) );
336  }
337  }
338 
339  return selectedBoundaries_int;
